scala pattern match on Array and List

Pattern match on Array. val arr = Array(1,2,3,4,5) arr match { case Array() => println("empty array") case Array(head, _*) => println(head) } arr match { case Array() => println("empty array") case Array(_, tail @ _*) => println(tail.mkString(",")) } Pattern match on List. val list = List(1,2,3,4,5) list match { case Nil => println("empty list") case head::tail => println(s"head:$head, tail:${tail.mkString(",")}") }
